##### Control #####

    flashcrowd flashcrowd

}

{
    protocol dht
    step SECOND
    from 1000

}

##### Network Churn #####
control.flashcrowd Turbulence
{
    protocol dht
    transport tr
    maxsize 500
    max_change 30
    p_rem 0
    p_add 1

    step SECOND

    init.0 CustomDist
    {
        dht dht
    }
    init.1 InitNode
    {
        protocol flashcrowd
        dht dht
    }
}

control.handler FlashcrowdController
{
    step SECOND
    dht dht
    protocol flashcrowd
}

control.churn Turbulence
{
    protocol dht
    transport tr
    minsize 3000
    delay_start 10 * SECOND
    max_change 10
    p_rem 1
    p_add 0
    step SECOND
}

control.swarm_trigger DiffusionTrigger
{
    protocol swarm
    delay_start 10 * SECOND
    step SECOND
}
